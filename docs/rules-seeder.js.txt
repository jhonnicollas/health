#!/usr/bin/env node
/**
 * HL Health Companion rules seeder
 *
 * Purpose:
 * - Read Master_Data_Health_Metrics.csv
 * - Normalize legacy CSV columns into HL_metricRules rows
 * - Generate an idempotent SQL seed file
 * - Optionally apply it to Cloudflare D1 using wrangler
 *
 * No external npm dependencies required.
 *
 * Usage:
 *   node rules-seeder.js
 *   node rules-seeder.js --input ./Master_Data_Health_Metrics.csv --output ./seed-rules.generated.sql
 *   node rules-seeder.js --input ./Master_Data_Health_Metrics.csv --output ./seed-rules.generated.sql --apply
 *   node rules-seeder.js --database multi_Ai_db --apply
 *
 * Notes:
 * - This script assumes the schema has already been applied.
 * - Table naming follows HL_*.
 * - Field naming follows camelCase.
 */

const fs = require('node:fs');
const path = require('node:path');
const { spawnSync } = require('node:child_process');

const args = parseArgs(process.argv.slice(2));
const inputPath = args.input || './Master_Data_Health_Metrics.csv';
const outputPath = args.output || './seed-rules.generated.sql';
const databaseName = args.database || 'multi_Ai_db';
const shouldApply = Boolean(args.apply);

function main() {
  if (!fs.existsSync(inputPath)) {
    exitWithError(`Input CSV tidak ditemukan: ${inputPath}`);
  }

  const csvText = fs.readFileSync(inputPath, 'utf8');
  const rows = parseLegacyCsv(csvText);
  const rules = rows
    .map((row, index) => normalizeCsvRule(row, index + 1))
    .filter(Boolean);

  const supplementedRules = mergeRules(rules, getSupplementalRules());
  const sql = buildRulesSql(supplementedRules);

  fs.writeFileSync(outputPath, sql, 'utf8');
  console.log(`Generated ${supplementedRules.length} rules -> ${outputPath}`);

  if (shouldApply) {
    applyWithWrangler(databaseName, outputPath);
  }
}

function parseArgs(argv) {
  const result = {};

  for (let i = 0; i < argv.length; i += 1) {
    const token = argv[i];

    if (token === '--apply') {
      result.apply = true;
      continue;
    }

    if (token.startsWith('--')) {
      const key = token.slice(2);
      const value = argv[i + 1];
      if (!value || value.startsWith('--')) {
        exitWithError(`Argument ${token} membutuhkan value.`);
      }
      result[key] = value;
      i += 1;
    }
  }

  return result;
}

function parseLegacyCsv(csvText) {
  const lines = csvText
    .replace(/^\uFEFF/, '')
    .split(/\r?\n/)
    .map((line) => line.trim())
    .filter(Boolean);

  const dataLines = lines.slice(1);

  return dataLines.map((line) => {
    const parts = splitCsvLineLoose(line);

    return {
      category: clean(parts[0]),
      indicator: clean(parts[1]),
      deviceName: clean(parts[2]),
      sex: clean(parts[3]),
      minValue: clean(parts[4]),
      maxValue: clean(parts[5]),
      unit: clean(parts[6]),
      status: clean(parts[7]),
      recommendation: clean(parts.slice(8).join(',')).replace(/,+$/g, '').trim()
    };
  });
}

function splitCsvLineLoose(line) {
  const cells = [];
  let current = '';
  let quoted = false;

  for (let i = 0; i < line.length; i += 1) {
    const char = line[i];
    const next = line[i + 1];

    if (char === '"' && quoted && next === '"') {
      current += '"';
      i += 1;
      continue;
    }

    if (char === '"') {
      quoted = !quoted;
      continue;
    }

    if (char === ',' && !quoted) {
      cells.push(current);
      current = '';
      continue;
    }

    current += char;
  }

  cells.push(current);
  return cells;
}

function normalizeCsvRule(row, rowNumber) {
  const metricCode = mapIndicator(row.indicator);

  if (!metricCode) {
    console.warn(`Skip row ${rowNumber}: indicator tidak dikenali -> ${row.indicator}`);
    return null;
  }

  const minValue = Number(row.minValue);
  const maxValue = Number(row.maxValue);

  if (!Number.isFinite(minValue) || !Number.isFinite(maxValue)) {
    console.warn(`Skip row ${rowNumber}: batas nilai tidak valid.`);
    return null;
  }

  const sex = sexMap.get(row.sex) || 'all';
  const unit = unitMap.get(row.unit) || row.unit;
  const status = row.status || 'Belum Ada Status';
  const severity = inferSeverity(metricCode, status, minValue, maxValue);
  const emergencyLevel = inferEmergencyLevel(metricCode, severity, minValue, maxValue);
  const id = makeRuleId(metricCode, sex, status, minValue, maxValue);
  const popupTitle = makePopupTitle(metricCode, status);
  const popupMessage = makePopupMessage(metricCode, status);
  const recommendation = row.recommendation || makeFallbackRecommendation(severity);

  return {
    id,
    metricCode,
    sex,
    ageMin: 0,
    ageMax: 200,
    minValue,
    maxValue,
    unit,
    status,
    severity,
    popupTitle,
    popupMessage,
    recommendation,
    sourceLabel: 'CSV internal',
    emergencyLevel,
    rulePriority: severity === 'emergency' ? 1 : 100,
    active: 1
  };
}

function mapIndicator(indicator) {
  const normalized = normalizeKey(indicator);

  if (indicatorMap.has(normalized)) {
    return indicatorMap.get(normalized);
  }

  if (normalized.includes('spo2')) return 'spo2';
  if (normalized.includes('heart')) return 'heartRate';
  if (normalized.includes('sistolik')) return 'systolic';
  if (normalized.includes('diastolik')) return 'diastolic';
  if (normalized.includes('puasa')) return 'glucoseFasting';
  if (normalized.includes('2 jam')) return 'glucosePostMeal';
  if (normalized.includes('kolesterol')) return 'cholesterolTotal';
  if (normalized.includes('uric') || normalized.includes('asam urat')) return 'uricAcid';
  if (normalized.includes('bmi')) return 'bmi';

  return null;
}

function inferSeverity(metricCode, status, minValue, maxValue) {
  const s = normalizeKey(status);

  if (s.includes('berat') || s.includes('krisis') || s.includes('darurat')) return 'emergency';
  if (s.includes('demam tinggi')) return 'critical';
  if (s.includes('diabetes') || s.includes('tinggi') || s.includes('tahap 2') || s.includes('obesitas')) return 'high';
  if (s.includes('prediabetes') || s.includes('pra') || s.includes('ringan') || s.includes('underweight') || s.includes('overweight')) return 'warning';
  if (s.includes('rendah')) return metricCode === 'glucoseFasting' || metricCode === 'bodyTemperature' ? 'critical' : 'info';
  if (s.includes('normal') || s.includes('optimal')) return 'normal';
  return 'info';
}

function inferEmergencyLevel(metricCode, severity, minValue, maxValue) {
  if (severity === 'emergency') return 'emergency';
  if (severity === 'critical') return 'urgent';
  if (severity === 'high') return 'urgent';
  if (severity === 'warning') return 'watch';
  return 'none';
}

function makeRuleId(metricCode, sex, status, minValue, maxValue) {
  const statusSlug = normalizeKey(status)
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-|-$/g, '')
    .slice(0, 32);

  const minSlug = String(minValue).replace('.', 'p');
  const maxSlug = String(maxValue).replace('.', 'p');
  return `rule-${metricCode}-${sex}-${statusSlug}-${minSlug}-${maxSlug}`;
}

function makePopupTitle(metricCode, status) {
  const metricName = metricDisplayName(metricCode);
  return `${metricName} ${status}`;
}

function makePopupMessage(metricCode, status) {
  return `${metricDisplayName(metricCode)} berada pada status ${status}. Gunakan hasil ini sebagai catatan dan verifikasi ulang angka pada alat.`;
}

function makeFallbackRecommendation(severity) {
  if (severity === 'emergency') {
    return 'Segera cek ulang. Jika nilai tetap kritis atau ada gejala berat, cari bantuan medis.';
  }
  if (severity === 'critical' || severity === 'high') {
    return 'Cek ulang setelah istirahat dan konsultasikan ke tenaga medis jika hasil berulang.';
  }
  if (severity === 'warning') {
    return 'Pantau hasil dan perbaiki pola hidup secara bertahap.';
  }
  return 'Pertahankan kebiasaan sehat dan catat hasil secara rutin.';
}

function rule(id, metricCode, sex, minValue, maxValue, unit, status, severity, popupTitle, popupMessage, recommendation, sourceLabel, emergencyLevel, rulePriority) {
  return {
    id,
    metricCode,
    sex,
    ageMin: 0,
    ageMax: 200,
    minValue,
    maxValue,
    unit,
    status,
    severity,
    popupTitle,
    popupMessage,
    recommendation,
    sourceLabel,
    emergencyLevel,
    rulePriority,
    active: 1
  };
}

function mergeRules(csvRules, supplementalRules) {
  const byKey = new Map();

  for (const item of [...csvRules, ...supplementalRules]) {
    const key = `${item.metricCode}|${item.sex}|${item.minValue}|${item.maxValue}|${item.status}`;
    if (!byKey.has(key)) {
      byKey.set(key, item);
    }
  }

  return Array.from(byKey.values()).sort((a, b) => {
    if (a.metricCode !== b.metricCode) return a.metricCode.localeCompare(b.metricCode);
    if (a.sex !== b.sex) return a.sex.localeCompare(b.sex);
    return a.minValue - b.minValue;
  });
}

function buildRulesSql(rules) {
  const lines = [];

  lines.push('-- Generated by rules-seeder.js');
  lines.push('-- Safe to run repeatedly.');
  lines.push('PRAGMA foreign_keys = ON;');
  lines.push('BEGIN TRANSACTION;');
  lines.push('');
  lines.push('INSERT OR IGNORE INTO HL_metricRules');
  lines.push('(id, metricCode, sex, ageMin, ageMax, minValue, maxValue, unit, status, severity, popupTitle, popupMessage, recommendation, sourceLabel, emergencyLevel, rulePriority, active)');
  lines.push('VALUES');

  const valueLines = rules.map((item) => `(${[
    sqlString(item.id),
    sqlString(item.metricCode),
    sqlString(item.sex),
    item.ageMin,
    item.ageMax,
    item.minValue,
    item.maxValue,
    sqlString(item.unit),
    sqlString(item.status),
    sqlString(item.severity),
    sqlString(item.popupTitle),
    sqlString(item.popupMessage),
    sqlString(item.recommendation),
    sqlString(item.sourceLabel),
    sqlString(item.emergencyLevel),
    item.rulePriority,
    item.active
  ].join(', ')})`);

  lines.push(valueLines.join(',\n') + ';');
  lines.push('');
  lines.push("INSERT OR IGNORE INTO HL_schemaMigrations (id, migrationName) VALUES ('seed-rules-generated', 'GeneratedMetricRulesSeed');");
  lines.push('COMMIT;');
  lines.push('');

  return lines.join('\n');
}

function applyWithWrangler(database, file) {
  console.log(`Applying ${file} to D1 database ${database}...`);

  const result = spawnSync('wrangler', ['d1', 'execute', database, '--file', file], {
    stdio: 'inherit',
    shell: process.platform === 'win32'
  });

  if (result.error) {
    exitWithError(`Gagal menjalankan wrangler: ${result.error.message}`);
  }

  if (result.status !== 0) {
    process.exit(result.status || 1);
  }
}

  return names[metricCode] || metricCode;
}

function normalizeKey(value) {
  return clean(value).toLowerCase().replace(/\s+/g, ' ').trim();
}

function clean(value) {
  return String(value ?? '').replace(/^"|"$/g, '').trim();
}

function sqlString(value) {
  if (value === null || value === undefined) return 'NULL';
  return `'${String(value).replace(/'/g, "''")}'`;
}

function exitWithError(message) {
  console.error(message);
  process.exit(1);
}
